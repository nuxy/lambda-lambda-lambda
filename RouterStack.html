Example
const func1 = function(req, res, next) {
    if (req.method() === 'POST') {
      res.status(405).send();
    } else {
      next();
    }
  };

  setFuncName(func1, 'middleware');
  stack.add(func1);

    ..

  const func2 = function(req, res) {
    res.setHeader('Content-Type', 'text/html');
    res.status(200).send('Hello World');
  };

  setFuncName(func2, 'route:get');
  stack.add(func2);
